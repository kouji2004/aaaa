2⃣event_notices controllerに以下を記述する
```ruby:app/controller/event_notices_control
def new
    @group = Group.find(params[:group_id])
  end
  
  def create
    
    @group = Group.find(params[:group_id])
    @title = params[:title]
    @body = params[:body] 
    
    event = { 
      :group => @group, 
      :title => @title, 
      :body => @body
      
    }
    
    EventMailer.send_notifications_to_group(event)
    
    render :sent
  end
  
  def sent
    redirect_to group_path(params[:group_id])
  endknowledge.'
```
3⃣route.rbに以下を追加する
```ruby:route.rb
  resources :groups, only: [:new, :index, :show, :create, :edit, :update] do
    resource :group_users, only: [:create, :destroy]
    resources :event_notices, only: [:new, :create]
    get "event_notices" => "event_notices#sent"
  end
```

9⃣mailersの編集
　　　　↓
　　app/mailers
　　　　↓
event_mailer.rbを開く
```ruby:app/mailers/event_mailer.rb
 def send_notification(member, event)
    @group = event[:group]
    @title = event[:title]
    @body = event[:body]

    @mail = EventMailer.new()
    mail(
      from: ENV['MAIL_ADDRESS'],
      to:   member.email,
      subject: 'New Event Notice!!'
    )
  end

  def self.send_notifications_to_group(event)
    group = event[:group]
    group.users.each do |member|
      EventMailer.send_notification(member, event).deliver_now
    end
  end
```
